# Modal

export const DefaultTemplate = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { Modal },
  methods: {
    action() {
      console.log('button pressed!')
    },
  },
  template: `
    <Modal v-bind="$props" @close="onClose">
      <button @click="action">button</button><br />
      <a href="/">link</a>
    </Modal>
  `,
});

export const ScrollTemplate = (args, { argTypes }) => ({
  props: Object.keys(argTypes),
  components: { Modal },
  data() {
    return {
      longTextHtml: 'scroll<br />'.repeat(50),
    }
  },
  methods: {
    action() {
      console.log('button pressed!')
    },
  },
  template: `
        <div>
          <span v-html="longTextHtml"></span>
          <br /><br />
          <Modal v-bind="$props" @close="onClose">
            <span v-html="longTextHtml"></span>
          </Modal>
          <br /><br />
          <span v-html="longTextHtml"></span>
        </div>
  `,
});
